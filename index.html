// ===== Gallery Auto Slide ===== let galleryIndex = 0; const gallery = document.querySelectorAll('.gallery-slider img'); function gallerySlide() { gallery.forEach(img => img.style.display = 'none'); gallery[galleryIndex].style.display = 'block'; galleryIndex = (galleryIndex + 1) % gallery.length; } gallerySlide(); setInterval(gallerySlide, 3000); // ===== Testimonials Auto Slide ===== let testIndex = 0; const testimonials = document.querySelectorAll('.testimonial-card'); function showTestimonial(index) { testimonials.forEach(t => t.classList.remove('active')); testimonials[index].classList.add('active'); } function testimonialSlide() { testIndex = (testIndex + 1) % testimonials.length; showTestimonial(testIndex); } testimonialSlide(); setInterval(testimonialSlide, 4000); // ===== Contact Form Success Message ===== const form = document.getElementById('contactForm'); const successMsg = document.getElementById('successMessage'); form.addEventListener('submit', function(e){ e.preventDefault(); fetch(form.action, { method:'POST', body: new FormData(form), headers: { 'Accept':'application/json' } }).then(response => { if(response.ok){ successMsg.style.display='block'; form.reset(); setTimeout(() => successMsg.style.display='none', 5000); } else { alert('Oops! Something went wrong.'); } }).catch(() => alert('Network error. Please try again later.')); });
